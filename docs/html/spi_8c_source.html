51 int spi_init() {

54  if (fd < 0)
55  send_error("Erro ao abrir o dispositivo");
56 
57  mode = SPI_MODE_0;
58 
59  ret = ioctl(fd, SPI_IOC_WR_MODE, &mode);

61  send_error("Erro ao setar o modo do SPI");
