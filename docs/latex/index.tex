\item include
\begin{DoxyItemize}
\item Header Files.
\end{DoxyItemize}

\begin{DoxyItemize}
\item Intel Galileo {\ttfamily /etc/init.d} files to setup pins.
\end{DoxyItemize}
\item lib

\item scripts
